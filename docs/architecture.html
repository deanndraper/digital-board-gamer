YouTube Channel Scanner Pipeline

flowchart LR
    subgraph INPUT["Input"]
        CH["YouTube Channel URLs\nfrom YAML config"]
    end

    subgraph SCRAPE["Phase 1: Scrape"]
        ST["scrapetube\nEnumerate videos"]
        DB_INS["Insert new videos\nstatus = PENDING"]
    end

    subgraph FILTER["Phase 2: Filter"]
        KW["Keyword skip\nchannel update, vlog, etc."]
        PRE["Claude Haiku\nPrefilter YES/NO"]
    end

    subgraph EXTRACT["Phase 3: Extract"]
        TR["youtube-transcript-api\nFetch transcript"]
        TRIM["Trim transcript\nFirst 1K + last 11K words"]
        CLAUDE["Claude Sonnet\nStructured JSON extraction"]
    end

    subgraph EXPORT["Phase 4: Export"]
        JSONL["channel_insights_*.jsonl\nAll COMPLETED records"]
        STATUS["Print status summary"]
    end

    CH --> ST --> DB_INS
    DB_INS --> KW
    KW -->|"SKIPPED"| DB_INS
    KW -->|"passed"| PRE
    PRE -->|"REJECTED"| DB_INS
    PRE -->|"YES"| TR
    TR --> TRIM --> CLAUDE
    CLAUDE -->|"COMPLETED"| JSONL
    CLAUDE -->|"FAILED_LLM"| DB_INS
    TR -->|"FAILED_TRANSCRIPT"| DB_INS
    JSONL --> STATUS

BGA Scraper Pipeline

flowchart LR
    subgraph LIST["Phase 1: Game List"]
        GL["GET boardgamearena.com/gamelist"]
        PARSE["Parse game_list JSON\n+ game_tags mapping"]
        SORT["Sort by games_played DESC\nTake top 200"]
        INS["Insert into bga_games\nwith tags, metrics"]
    end

    subgraph DETAIL["Phase 2: Detail Pages"]
        GP["GET gamepanel?game=ID\nfor each PENDING game"]
        ROW["Parse row-data divs\ndesigner, publisher, year,\ncomplexity, strategy, luck,\ninteraction, duration, players"]
        IMG["Extract box image URL"]
    end

    subgraph BGG["Phase 3: BGG Enrichment"]
        BGGF["GET boardgamegeek.com/boardgame/ID"]
        BGGP["Parse geekitemPreload JSON\nrating, weight, rank,\nnum_ratings, num_owned"]
    end

    subgraph OUT["Output"]
        DB["bga_games.db\n200 games, 44+ fields"]
        JL["bga_games.jsonl"]
        XLSX["BGA_Top_200_Games.xlsx"]
    end

    GL --> PARSE --> SORT --> INS
    INS --> GP --> ROW --> IMG
    IMG --> BGGF --> BGGP
    BGGP --> DB
    DB --> JL
    DB --> XLSX

Publish Pipeline (publish.sh)

flowchart LR
    PULL["git pull"] --> ENV["source .env\nsource .venv/bin/activate"]
    ENV --> RUN1["python llm_orchestrator.py\n+ --retry\n(default 3 channels)"]
    RUN1 --> RUN2["SCANNER_CONFIG=boardgameco\npython llm_orchestrator.py\n+ --retry"]
    RUN2 --> GEN["python generate_site.py\nReads all channel_insights_*.jsonl"]
    GEN --> DIFF{"Changes\ndetected?"}
    DIFF -->|"No"| DONE["Done"]
    DIFF -->|"Yes"| COMMIT["git add + commit + push"]
    COMMIT --> PAGES["GitHub Pages\nauto-deploys"]
